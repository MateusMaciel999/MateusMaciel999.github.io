/* No final do HTML */ function toggleFavorito(botao) { const img = botao.querySelector('img'); const isFavorito = botao.getAttribute('data-favorito') === 'true'; if (!isFavorito) { img.src = "img(Fotos)/heart-red.png" ; botao.setAttribute('data-favorito', 'true'); } else { img.src = "img(Fotos)/heart-white.png"; botao.setAttribute('data-favorito', 'false'); } }
